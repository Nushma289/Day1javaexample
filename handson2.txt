import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

public class Hand1 {

	public static void main(String[] args) {
		List<Integer> matches=new ArrayList<>();
		Scanner s=new Scanner(System.in);
		System.out.println("enter no.of matches");
		int n=s.nextInt();
		for(int j=0;j<n;j++) {
			matches.add(s.nextInt());
		}
		int sum=0;
		for(int j=0;j<n;j++) {
			sum +=matches.get(j);
		}
		System.out.println(sum);
		float avg=(float)sum/n;
		System.out.println(avg);

	}

}

4.package com.day8;

import java.util.HashMap;
import java.util.Iterator;
import java.util.Map;
import java.util.Scanner;
import java.util.Set;
class UserMainCode{
	static double calculateAverage(double sum,int count) {
		return (double)sum/count;
		
	}
	
}

public class Hand1 {

	public static void main(String[] args) {
		Map<Integer,Double> m=new HashMap<>();
		Scanner sc=new Scanner(System.in);
		System.out.println("enter s value");
		int s=sc.nextInt();
		double sum=0;
		int count=0;
		System.out.println("enter key value pairs");
		for(int i=0;i<s;i++) {
			m.put(sc.nextInt(), sc.nextDouble());
		}
		Set s1=m.keySet();
		Iterator itr=s1.iterator();
		while(itr.hasNext()) {
			Integer i=(Integer)itr.next();
			if(i%2!=0) {
				sum +=m.get(i);
				count++;
			}
		}
		UserMainCode uc=new UserMainCode();
		double avg=uc.calculateAverage(sum, count);
		System.out.printf("%.2f",avg);

	}


}
